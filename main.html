<!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Vimeo 4K Video</title>

    <script src="https://player.vimeo.com/api/player.js"></script>

</head>

<body>

    <iframe

        id="vimeo-player"

        src="https://player.vimeo.com/video/959000858?badge=0&autopause=0&player_id=0&app_id=58479"

        width="3840"

        height="2160"

        frameborder="0"

        allow="autoplay; fullscreen; picture-in-picture; clipboard-write"

        title="My Edit"

    ></iframe>


    <script>

        document.addEventListener('DOMContentLoaded', function() {

            const iframe = document.getElementById('vimeo-player');

            const player = new Vimeo.Player(iframe);


            // Set the quality to 4K (2160p)

            player.setQuality('2160p').then(function(quality) {

                console.log('Quality set to: ' + quality);

            }).catch(function(error) {

                console.error('Error setting quality:', error.name);

            });


            // Automatically request fullscreen

            player.on('play', function() {

                player.requestFullscreen().catch(function(error) {

                    console.error('Error requesting fullscreen:', error.name);

                });

            });


            // Optionally, you can also listen for quality change events

            player.on('qualitychange', function(event) {

                console.log('Quality changed to: ' + event.quality);

            });


            // Start playing the video automatically

            player.play().catch(function(error) {

                console.error('Error playing video:', error.name);

            });

        });

    </script>

</body>

</html>
